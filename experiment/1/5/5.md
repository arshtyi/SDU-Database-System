```sql
CREATE TABLE test5_01 (
    bid CHAR(4),
    bname VARCHAR2(20),
    dcount NUMBER(10),
    pcount NUMBER(10),
    sum_amount NUMBER(10,2),
    avg_amount NUMBER(10,2)
);
INSERT INTO test5_01
SELECT
    b.bid,
    b.bname,
    COUNT(d.pid),
    COUNT(DISTINCT dp.pid),
    SUM(d.amount),
    ROUND(AVG(d.amount), 2)
FROM bk.bank b
JOIN bk.deposit d ON b.bid = d.bid
JOIN bk.deptor dp ON d.pid = dp.pid
GROUP BY b.bid, b.bname;
```

## 2

```sql
CREATE TABLE test5_02 (
    city VARCHAR2(10),
    bid CHAR(4),
    bname VARCHAR2(20),
    dcount NUMBER(10),
    sum_amount NUMBER(30,2)
);
INSERT INTO test5_02
SELECT
    b.city,
    NVL(b.bid, '0000') AS bid,
    NVL(b.bname, '合计') AS bname,
    COUNT(d.pid) AS dcount,
    SUM(d.amount) AS sum_amount
FROM bk.bank b
JOIN bk.deposit d ON b.bid = d.bid
GROUP BY b.city, ROLLUP((b.bid, b.bname))
ORDER BY b.city, bid;
```

## 3

```sql
CREATE TABLE test5_03 (
    generation CHAR(6),
    dcount NUMBER(10),
    sum_amount NUMBER(30,2),
    total_count NUMBER(10),
    count_percent NUMBER(30,2),
    total_amount NUMBER(30,2),
    amount_percent NUMBER(30,2)
);
INSERT INTO test5_03
WITH GenStats AS (
    SELECT
        TO_CHAR(TRUNC(EXTRACT(YEAR FROM de.birthdate), -1)) || '后' AS generation,
        COUNT(dp.pid) AS dcount,
        SUM(dp.amount) AS sum_amount
    FROM bk.deptor de
    JOIN bk.deposit dp ON de.pid = dp.pid
    GROUP BY TRUNC(EXTRACT(YEAR FROM de.birthdate), -1)
)
SELECT
    generation,
    dcount,
    sum_amount,
    SUM(dcount) OVER() AS total_count,
    ROUND(dcount * 100 / SUM(dcount) OVER(), 2) AS count_percent,
    SUM(sum_amount) OVER() AS total_amount,
    ROUND(sum_amount * 100 / SUM(sum_amount) OVER(), 2) AS amount_percent
FROM GenStats;
```

```sql
CREATE TABLE test5_04 (
    City VARCHAR2(10),
    dcount1 NUMBER(10),
    sum_amount1 NUMBER(30,2),
    amount_percent1 NUMBER(30,2),
    dcount2 NUMBER(10),
    sum_amount2 NUMBER(30,2),
    amount_percent2 NUMBER(30,2),
    dcount NUMBER(10),
    sum_amount NUMBER(30,2)
);
INSERT INTO test5_04
WITH CityStats AS (
    SELECT
        b.city,
        COUNT(CASE WHEN b.stru LIKE '%国有%' THEN d.pid END) AS dcount1,
        NVL(SUM(CASE WHEN b.stru LIKE '%国有%' THEN d.amount END), 0) AS sum_amount1,
        COUNT(CASE WHEN b.stru LIKE '%股份%' THEN d.pid END) AS dcount2,
        NVL(SUM(CASE WHEN b.stru LIKE '%股份%' THEN d.amount END), 0) AS sum_amount2,
        COUNT(d.pid) AS dcount,
        NVL(SUM(d.amount), 0) AS sum_amount
    FROM bk.bank b
    JOIN bk.deposit d ON b.bid = d.bid
    GROUP BY b.city
)
SELECT
    city,
    dcount1,
    sum_amount1,
    ROUND(sum_amount1 * 100 / NULLIF(sum_amount, 0), 2) AS amount_percent1,
    dcount2,
    sum_amount2,
    ROUND(sum_amount2 * 100 / NULLIF(sum_amount, 0), 2) AS amount_percent2,
    dcount,
    sum_amount
FROM CityStats;
```

## 6

```sql
CREATE OR REPLACE VIEW test5_06 AS
WITH PersonBankSum AS (
    SELECT
        dp.bid AS bid,
        de.pid AS pid,
        SUM(dp.amount) AS sum_amount
    FROM bk.deposit dp
    JOIN bk.deptor de ON dp.pid = de.pid
    GROUP BY dp.bid, de.pid
),
RankedSum AS (
    SELECT
        bid,
        pid,
        sum_amount,
        RANK() OVER (PARTITION BY bid ORDER BY sum_amount DESC) AS rnk
    FROM PersonBankSum
)
SELECT
    bid,
    pid,
    sum_amount
FROM RankedSum
WHERE rnk = 1;
```

```sql
CREATE OR REPLACE VIEW test5_08 AS
SELECT
    b.city AS CITY,
    TO_CHAR(TRUNC(EXTRACT(YEAR FROM de.birthdate), -1)) || '后' AS GENERATION,
    EXTRACT(YEAR FROM dp.dtime) AS DYEAR,
    COUNT(dp.pid) AS DCOUNT,
    SUM(dp.amount) AS SUM_AMOUNT
FROM bk.bank b
JOIN bk.deposit dp ON b.bid = dp.bid
JOIN bk.deptor de ON dp.pid = de.pid
GROUP BY
    b.city,
    TO_CHAR(TRUNC(EXTRACT(YEAR FROM de.birthdate), -1)) || '后',
    EXTRACT(YEAR FROM dp.dtime)
ORDER BY
    CITY,
    GENERATION,
    DYEAR;
```

## 9

```sql
CREATE OR REPLACE VIEW test5_09 AS
SELECT
    de.pid AS PID,
    de.pname AS PNAME,
    de.sex AS SEX,
    COUNT(dp.pid) AS DCOUNT,
    SUM(dp.amount) AS SUM_AMOUNT
FROM bk.deptor de
JOIN bk.deposit dp ON de.pid = dp.pid
WHERE de.sex = '女' AND dp.amount > 2000
GROUP BY de.pid, de.pname, de.sex
HAVING COUNT(dp.pid) > 130 AND SUM(dp.amount) < 900000;
```

## 10

```sql
CREATE OR REPLACE VIEW test5_10 AS
SELECT
    TO_CHAR(TRUNC(EXTRACT(YEAR FROM de.birthdate), -1)) || '后' AS GENERATION,
    COUNT(de.pid) AS PCOUNT,
    SUM(t.total_amount) AS SUM_AMOUNT
FROM bk.deptor de
JOIN (
    SELECT pid, SUM(amount) AS total_amount
    FROM bk.deposit
    GROUP BY pid
    HAVING SUM(amount) > 500000
) t ON de.pid = t.pid
GROUP BY TRUNC(EXTRACT(YEAR FROM de.birthdate), -1)
ORDER BY GENERATION;
```
