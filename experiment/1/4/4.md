```sql
SET

            WHEN sex LIKE '%男%' OR UPPER(sex) IN ('M', '1') THEN '男'

            ELSE NULL
            END,
```

```sql
CREATE TABLE test4_02 AS
SELECT * FROM bk.deptor3;
UPDATE test4_02
SET
    age = CASE
            WHEN (2019 - TO_NUMBER(TO_CHAR(birthdate, 'YYYY'))) < age
            THEN (2019 - TO_NUMBER(TO_CHAR(birthdate, 'YYYY')))
            ELSE age
            END,
    pid = CASE
                WHEN SUBSTR(pid, 7, 8) <> TO_CHAR(birthdate, 'YYYYMMDD')
                THEN SUBSTR(pid, 1, 6) || TO_CHAR(birthdate, 'YYYYMMDD') || SUBSTR(pid, 15)
                ELSE pid
            END
    WHERE (2019 - TO_NUMBER(TO_CHAR(birthdate, 'YYYY'))) < age
        OR SUBSTR(pid, 7, 8) <> TO_CHAR(birthdate, 'YYYYMMDD');
```

## 3

```sql
CREATE TABLE test4_03 AS
SELECT * FROM bk.deptor3;
ALTER TABLE test4_03 ADD (
    total_amount1 NUMBER(10,2),
    total_amount2 NUMBER(10,2),
    total_amount NUMBER(10,2)
);
UPDATE test4_03 t
SET
    total_amount2 = NVL((SELECT SUM(amount) FROM bk.deposit d WHERE d.pid = t.pid), 0),
    total_amount1 = NVL((SELECT SUM(amount) FROM bk.deposit d WHERE d.pid = t.parentpid), 0);
UPDATE test4_03
SET total_amount = total_amount1 + total_amount2;
DELETE FROM test4_03
WHERE rowid NOT IN (
    SELECT rid
    FROM (
        SELECT rowid AS rid
        FROM test4_03
        ORDER BY total_amount DESC
    )
    WHERE rownum <= 20
);
```

```sql
CREATE TABLE test4_04 AS
SELECT * FROM bk.deptor3;
ALTER TABLE test4_04 ADD (
    total_count NUMBER,
    total_amount NUMBER(15,2),
    avg_amount NUMBER(15,2)
);
UPDATE test4_04 t
    SET (total_count, total_amount, avg_amount) = (
        SELECT COUNT(*), SUM(amount), ROUND(AVG(amount), 2)
        FROM bk.deposit d
        WHERE d.pid = t.pid
    )
    WHERE EXISTS (
        SELECT 1
        FROM bk.deposit d
        WHERE d.pid = t.pid
    );
```

## 7

```sql
CREATE TABLE test4_07 AS
SELECT * FROM bk.deptor3;
ALTER TABLE test4_07 ADD (
    total_interest NUMBER(30,2)
);
UPDATE test4_07 t
SET total_interest = (
    SELECT SUM(ROUND(d.amount * (TRUNC(SYSDATE, 'dd') - TRUNC(d.dtime, 'dd')) * b.dailyrate, 2))
    FROM bk.deposit d
    JOIN bk.bank b ON d.bid = b.bid
    WHERE d.pid = t.pid
)
WHERE EXISTS (
    SELECT 1
    FROM bk.deposit d
    WHERE d.pid = t.pid
);
```

```sql
      AND t3.birthdate = (
          SELECT MAX(t2.birthdate)
```

```sql
          SELECT MAX(t2.birthdate)
```
