## 3

<!-- > "逃课"做法 -->

```sql
CREATE TABLE test3_03 AS
SELECT * FROM bk.deptor3;
DELETE FROM test3_03
WHERE sex IS NOT NULL
    AND sex NOT IN (
        SELECT sex FROM (
            SELECT sex FROM test3_03
            WHERE sex IS NOT NULL
            GROUP BY sex
            ORDER BY COUNT(*) DESC
        ) WHERE ROWNUM <= 2
    );
```

## 6

```sql
CREATE TABLE test3_06 AS
SELECT * FROM bk.deptor3;

DELETE FROM test3_06
WHERE NOT REGEXP_LIKE(pid, '^[0-9]+[Xx]?$')
    OR TO_CHAR(birthdate, 'YYYYMMDD') <> SUBSTR(pid, 7, 8)
    OR (sex IS NOT NULL AND sex NOT IN (
            SELECT sex FROM (
                SELECT sex FROM test3_06
                WHERE sex IS NOT NULL
                GROUP BY sex
                ORDER BY COUNT(*) DESC
            ) WHERE ROWNUM <= 2
        ))
    OR lengthb(pname) != length(pname) * 2
    OR age != 2019 - extract(year from birthdate);
```

## 9

```sql
CREATE TABLE test3_09 AS
SELECT * FROM bk.deptor3
WHERE sex IN ('男', '女');

DELETE FROM test3_09 A
WHERE NOT EXISTS (
    SELECT 1 FROM test3_09 B
    WHERE A.birthdate = B.birthdate
      AND A.sex <> B.sex
);
```
