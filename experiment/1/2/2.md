# Lab 1

## 1

```sql
CREATE OR REPLACE VIEW test2_01 AS
SELECT * FROM bk.deptor
WHERE pname NOT LIKE '高%'
    AND pname NOT LIKE '韩%'
    AND (pname LIKE '%建%' OR pname LIKE '%平%')
    AND EXTRACT(YEAR FROM birthdate) = 1971
ORDER BY birthdate DESC, sex, pname;
```

## 2

```sql
Create or replace view test2_02 as
select d.pid, d.pname, dp.did, b.bid, b.bname, dp.amount, dp.dtime, b.city
from bk.deptor d
join bk.deposit dp on d.pid = dp.pid
join bk.bank b on dp.bid = b.bid
where d.pname = '李龙'
    and dp.amount > 1000
    and b.city = '北京';
```

## 4

```sql
CREATE OR REPLACE VIEW test2_05 AS
SELECT
    c.pid AS childpid,
    c.pname AS childname,
    c.birthdate AS childbirthdate,
    p.pid AS parentpid,
    p.pname AS parentname,
    p.birthdate AS parentbirthdate
FROM bk.deptor c
JOIN bk.deptor p ON c.parentpid = p.pid
WHERE c.birthdate - p.birthdate > 8030;
```

## 6

```sql
CREATE OR REPLACE VIEW test2_06 AS
SELECT
    c.pid AS childpid,
    c.pname AS childname,
    c.birthdate AS childbirthdate,
    p.pid AS parentpid,
    p.pname AS parentname,
    p.birthdate AS parentbirthdate
FROM bk.deptor c
JOIN bk.deptor p ON c.parentpid = p.pid
WHERE c.birthdate - p.birthdate > 365 * 22
    AND c.pid IN (SELECT pid FROM bk.deposit WHERE amount > 6000)
    AND p.pid IN (SELECT pid FROM bk.deposit WHERE amount > 9000);
```

## 8

```sql
CREATE OR REPLACE VIEW test2_08 AS
SELECT * FROM bk.deptor
WHERE pid IN (
    SELECT pid FROM bk.deposit
    WHERE bid IN (SELECT bid FROM bk.bank WHERE city = '成都')
)
AND pid NOT IN (
    SELECT pid FROM bk.deposit
    WHERE bid IN (SELECT bid FROM bk.bank WHERE city = '西宁')
);
```

## 9

```sql
CREATE OR REPLACE VIEW test2_09 AS
SELECT * FROM bk.deptor
WHERE pid IN (
    SELECT pid FROM bk.deposit
    JOIN bk.bank ON bk.deposit.bid = bk.bank.bid
    WHERE amount > 5000 AND city = '成都'
)
AND pid IN (
    SELECT pid FROM bk.deposit
    JOIN bk.bank ON bk.deposit.bid = bk.bank.bid
    WHERE amount > 6000 AND city = '重庆'
);
```

## 10

### M1

```sql
CREATE OR REPLACE VIEW test2_10 AS
SELECT * FROM bk.deptor d
WHERE NOT EXISTS (
    SELECT * FROM bk.bank b
    WHERE NOT EXISTS (
        SELECT * FROM bk.deposit dp
        WHERE dp.pid = d.pid AND dp.bid = b.bid
    )
);
```

### M2

```sql
CREATE OR REPLACE VIEW test2_10_2 AS
SELECT * FROM bk.deptor
WHERE pid NOT IN (
    SELECT pid FROM (
        SELECT d.pid, b.bid FROM bk.deptor d CROSS JOIN bk.bank b
        MINUS
        SELECT pid, bid FROM bk.deposit
    )
);
```
