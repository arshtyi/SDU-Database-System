## 1

```sql
CREATE TABLE test7_01 AS
SELECT * FROM bk.deptor WHERE 1=2;
CREATE UNIQUE INDEX idx_test7_01_pid ON test7_01(pid);
INSERT INTO test7_01
SELECT * FROM bk.deptor
WHERE parentpid IN (
    SELECT parentpid FROM bk.deptor
    WHERE parentpid IS NOT NULL
    GROUP BY parentpid
    HAVING COUNT(*) >= 2
);
INSERT INTO test7_01
SELECT * FROM bk.deptor d
WHERE d.pid IN (
    SELECT parentpid FROM bk.deptor
    WHERE parentpid IS NOT NULL
    GROUP BY parentpid
    HAVING COUNT(*) >= 2
)
AND NOT EXISTS (
    SELECT 1 FROM test7_01 t WHERE t.pid = d.pid
);
```

## 3

```sql
CREATE TABLE test7_03 AS
SELECT * FROM bk.deptor WHERE 1=2;
ALTER TABLE test7_03 ADD total_amount NUMBER;
INSERT INTO test7_03
SELECT d.*, t.total_amount
FROM bk.deptor d
JOIN (
    SELECT pid, SUM(amount) AS total_amount
    FROM bk.deposit
    GROUP BY pid
    HAVING SUM(amount) > 2000000
) t ON d.pid = t.pid;
```
