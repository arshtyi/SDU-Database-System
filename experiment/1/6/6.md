```sql
create or replace view test6_01 as
select substr(pname, 2) as First_name, count(*) as frequency
from bk.deptor
group by substr(pname, 2)
having count(*) > 5;
```

## 2

```sql
create or replace view test6_02 as
select letter, count(*) as frequency
from (
    select substr(pname, 2, 1) as letter from bk.deptor
    union all
    select substr(pname, 3, 1) from bk.deptor
)
where letter is not null
group by letter
having count(*) > 10;
```

## 4

```sql
create or replace view test6_04 as
select de.pid, de.pname, de.sex, de.age, count(dp.pid) as count_amount, sum(dp.amount) as sum_amount
from bk.deptor de
join bk.deposit dp on de.pid = dp.pid
join bk.bank b on dp.bid = b.bid
where b.city = '北京' and de.sex = '女' and de.age > 30
group by de.pid, de.pname, de.sex, de.age
having sum(dp.amount) > 500000 and count(dp.pid) > 60;
```

## 5

### M1

```sql
create or replace view test6_05 as
select b.bid, b.bname,
    (select max(amount) from bk.deposit d1 where d1.bid = b.bid) as max_amount,
    (select count(*) from bk.deposit d2
     where d2.bid = b.bid
       and d2.amount = (select max(amount) from bk.deposit d3 where d3.bid = b.bid)
    ) as max_amount_count
from bk.bank b
where exists (select 1 from bk.deposit d where d.bid = b.bid);
```

### M2

```sql
create or replace view test6_05 as
with max_amt as (
    select bid, max(amount) as max_amount
    from bk.deposit
    group by bid
),
max_amt_cnt as (
    select d.bid, count(*) as max_amount_count
    from bk.deposit d
    join max_amt m on d.bid = m.bid and d.amount = m.max_amount
    group by d.bid
)
select b.bid, b.bname, m.max_amount, c.max_amount_count
from bk.bank b
join max_amt m on b.bid = m.bid
join max_amt_cnt c on b.bid = c.bid;
```

### M3

```sql
create or replace view test6_05 as
select b.bid, b.bname, t2.max_amount, t3.max_amount_count
from bk.bank b,
    (select bid, max(amount) as max_amount
     from bk.deposit
     group by bid) t2,
    (select d.bid, count(*) as max_amount_count
     from bk.deposit d
     join (select bid, max(amount) as max_amount from bk.deposit group by bid) m
       on d.bid = m.bid and d.amount = m.max_amount
     group by d.bid) t3
where b.bid = t2.bid and b.bid = t3.bid;
```

```sql
create or replace view test6_07 as
select * from bk.deptor d
where sex = '女'
and not exists (
    select 1 from bk.bank b
    where not exists (
        select 1 from bk.deposit dp
        where dp.pid = d.pid and dp.bid = b.bid and dp.amount > 200
    )
);
```

## 9

```sql
create or replace view test6_09 as
select * from bk.deptor
where age > 40 and sex = '女'
and pid in (
    select dp.pid
    from bk.deposit dp
    join bk.bank b on dp.bid = b.bid
    where b.city = '北京'
    group by dp.pid
    having sum(dp.amount) > 300000
)
and pid not in (
    select dp.pid
    from bk.deposit dp
    join bk.bank b on dp.bid = b.bid
    where b.city = '西宁'
);
```

## 10

```sql
create or replace view test6_10 as
select rank, pid, sum_amount
from (
    select rank() over(order by sum(amount) desc) as rank, pid, sum(amount) as sum_amount
    from bk.deposit
    group by pid
)
where rank <= 20;
```
